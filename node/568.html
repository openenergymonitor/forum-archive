I found it and here's the code:
 
 
#include <GLCD_ST7565.h>
#include <JeeLib.h>
 
const byte picture[16*64] PROGMEM = {
 
//here comes the HEX code of the bitmap
 
};
 
GLCD_ST7565 glcd;
 
void setup () {
  glcd.begin(0x18);
  glcd.backLight(255);
  
  glcd.drawBitmap(0, 0, picture, 128, 64, WHITE);
  
  glcd.refresh();
}
 
void loop () {}
