//Arduino Sketch for example program with data being sent to the arduino double valA,valB; int incomingByte; void setup() { Serial.begin(115200); } void loop() { if (Serial.available()>0) { incomingByte = Serial.read(); if (incomingByte=='A') valA=0.0; if (incomingByte=='B') valB=0.0; } valA=valA+0.1; valB=valB+0.2; //Print value A Serial.print(valA); //Then its letter identifier Serial.print("A"); //Print value B Serial.print(valB); //Then its letter identifier Serial.print("B"); delay(500); }
